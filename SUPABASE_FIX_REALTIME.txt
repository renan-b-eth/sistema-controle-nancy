-- 1. Habilitar o Realtime para a tabela entradas
-- O Realtime no Supabase por padrão vem desativado para novas tabelas.
-- Este comando inclui a tabela 'entradas' na publicação de tempo real.

BEGIN;
  -- Remove se já existir para evitar duplicidade
  DROP PUBLICATION IF EXISTS supabase_realtime;
  
  -- Cria a publicação para a tabela entradas
  CREATE PUBLICATION supabase_realtime FOR TABLE "entradas";
COMMIT;

-- 2. Garantir que a tabela envie todos os dados na mudança (CRÍTICO)
ALTER TABLE "entradas" REPLICA IDENTITY FULL;

-- 3. Mensagem de verificação
SELECT 'REALTIME ATIVADO COM SUCESSO' as status;
